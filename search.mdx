{/*
  This page hosts a client-side semantic search over the docs.

  - The index is built by `scripts/build-search-index.mjs` and shipped as
    `assets/search-index.js` (configured in `search.config.json`).
  - The embedding model is loaded from jsdelivr (cached by the browser).
  - Mintlify's snippet imports do not wire through in this CLI version, so
    the entire component is defined inline. React hooks (useState, useEffect,
    useRef) are available as globals inside MDX components.
*/}

export const cosine = (a, b) => {
  let s = 0;
  for (let i = 0; i < a.length; i++) s += a[i] * b[i];
  return s;
};

export const escapeRegex = (s) => s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');

export const highlightFn = (text, terms) => {
  if (!terms.length) return text;
  const pattern = new RegExp(
    '(' + terms.filter((t) => t.length >= 2).map(escapeRegex).join('|') + ')',
    'gi'
  );
  const parts = String(text).split(pattern);
  return parts.map((p, i) =>
    pattern.test(p) ? (
      <mark
        key={i}
        style={{
          background: 'rgba(22,163,74,0.25)',
          color: 'inherit',
          padding: '0 2px',
          borderRadius: 3,
        }}
      >
        {p}
      </mark>
    ) : (
      <span key={i}>{p}</span>
    )
  );
};

export const rankAgainst = (queryVec, chunks, topK, minScore) => {
  const scored = [];
  for (let i = 0; i < chunks.length; i++) {
    const score = cosine(queryVec, chunks[i].vector);
    if (score < minScore) continue;
    scored.push({ score, chunk: chunks[i] });
  }
  scored.sort((a, b) => b.score - a.score);
  const seen = new Set();
  const uniq = [];
  for (const r of scored) {
    if (seen.has(r.chunk.url)) continue;
    seen.add(r.chunk.url);
    uniq.push(r);
    if (uniq.length >= topK) break;
  }
  return uniq;
};

export const loadIndexGlobal = () => {
  if (typeof window === 'undefined') return Promise.reject(new Error('no window'));
  if (window.__VSEARCH_INDEX__) return Promise.resolve(window.__VSEARCH_INDEX__);
  return new Promise((resolve, reject) => {
    const ready = () => {
      if (window.__VSEARCH_INDEX__) resolve(window.__VSEARCH_INDEX__);
      else reject(new Error('index global missing'));
    };
    window.addEventListener('vsearch:index-loaded', ready, { once: true });
    if (document.querySelector('script[data-vsearch-index]')) return;
    const s = document.createElement('script');
    s.src = '/assets/search-index.js';
    s.async = true;
    s.dataset.vsearchIndex = 'true';
    s.onerror = () => reject(new Error('failed to load /assets/search-index.js'));
    document.head.appendChild(s);
  });
};

export const loadEmbedderFromCdn = (modelName, version) => {
  if (typeof window === 'undefined') return Promise.reject(new Error('no window'));
  const makePipeline = (t) => t.pipeline('feature-extraction', modelName);
  if (window.__VSEARCH_TRANSFORMERS__) return makePipeline(window.__VSEARCH_TRANSFORMERS__);
  const v = version || '2.17.2';
  return new Promise((resolve, reject) => {
    const ready = () => {
      const t = window.__VSEARCH_TRANSFORMERS__;
      if (!t) return reject(new Error('transformers global missing'));
      makePipeline(t).then(resolve, reject);
    };
    const fail = () => reject(new Error('failed to load @xenova/transformers from cdn.jsdelivr.net'));
    window.addEventListener('vsearch:transformers-loaded', ready, { once: true });
    window.addEventListener('vsearch:transformers-failed', fail, { once: true });
    if (document.querySelector('script[data-vsearch-transformers]')) return;
    const s = document.createElement('script');
    s.type = 'module';
    s.dataset.vsearchTransformers = 'true';
    // Inline body avoids putting an `import()` keyword in the MDX source —
    // Mintlify's static analyzer rejects non-local imports.
    s.textContent =
      "import('https://cdn.jsdelivr.net/npm/@xenova/transformers@" + v + "').then(" +
      "(t)=>{window.__VSEARCH_TRANSFORMERS__=t;" +
      "window.dispatchEvent(new Event('vsearch:transformers-loaded'));}," +
      "(e)=>{console.error('[vector-search] cdn import failed',e);" +
      "window.dispatchEvent(new Event('vsearch:transformers-failed'));});";
    document.head.appendChild(s);
  });
};

        return;
      }
      setStatus('Searching…');
      try {

        if (cancelled || mySeq !== seq) return;
        const queryVec = Array.from(out.data);
        const idx = indexRef;
        const ui = idx.ui || {};
        const topK = ui.topK || 8;
        const minScore = typeof ui.minScore === 'number' ? ui.minScore : 0.2;
        const ranked = rankAgainst(queryVec, idx.chunks, topK, minScore);
        if (cancelled || mySeq !== seq) return;
        const terms = trimmed.split(/\s+/).filter((t) => t.length >= 2);
        renderResults(ranked, terms);
        setStatus(ranked.length + ' result' + (ranked.length === 1 ? '' : 's'));
      } catch (err) {
        console.error('[vector-search] query failed', err);
        if (!cancelled) setStatus('Query failed');
      }
    };

    const onInput = (e) => {
      const v = e.target.value;
      syncOverlay();
      if (debounce) clearTimeout(debounce);
      debounce = setTimeout(() => runQuery(v), 120);
    };
    input.addEventListener('input', onInput);

    (async () => {
      try {
        setStatus('Loading index…');
        const index = await loadIndexGlobal();
        if (cancelled) return;
        indexRef = index;
        if (index.ui && index.ui.placeholder) placeholderText = index.ui.placeholder;

        setStatus('Loading model…');
        embedder = await loadEmbedderFromCdn(index.model, index.transformersVersion);
        if (cancelled) return;

        setEnabled(true, placeholderText);
        setStatus(index.chunks.length + ' chunks');
        try { input.focus(); } catch (_) {}
      } catch (err) {
        console.error('[vector-search]', err);
        if (cancelled) return;
        const msg = err && err.message ? err.message : String(err);
        setEnabled(false, 'Search unavailable');
        setStatus('Failed: ' + msg);
      }
    })();

    return () => {
      cancelled = true;
      input.removeEventListener('input', onInput);
      document.removeEventListener('keydown', onKeydown);
      if (debounce) clearTimeout(debounce);
      if (bar.parentNode) bar.parentNode.removeChild(bar);
      if (resultsHost.parentNode) resultsHost.parentNode.removeChild(resultsHost);
      if (overlay.parentNode) overlay.parentNode.removeChild(overlay);
      if (styleEl.parentNode) styleEl.parentNode.removeChild(styleEl);
    };
  }, []);

  // Reserve the visual space at the top of the page so content doesn't
  // sit underneath the body-level fixed bar.
  return <div aria-hidden="true" style={{ height: 88 }} />;
};
